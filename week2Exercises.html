This exercise is to determine if a number is even or odd


// Your code here.
function isEven(x)
{
if(x%2 == 0)
{
return true;
}
else if(x%2 == 1)
{
return false;
}
else
{
return "??"
}
}
console.log(isEven(50));
// → true
console.log(isEven(75));
// → false
